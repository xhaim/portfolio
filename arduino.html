TEMPERATURE BASED FAN SPEED CONTROLLER USING ARDUINO

A Temperature Based Fan Speed Controller using Arduino is a system that utilizes an Arduino microcontroller and an LM35 temperature sensor to adjust the speed of a fan based on the ambient temperature. The system measures the temperature with the LM35 sensor and then modulates the voltage of a PWM (Pulse Width Modulation) signal to control the speed of a DC fan. This type of system is commonly employed in electronic equipment or home automation systems to maintain desired temperature levels and enhance energy efficiency. The sensed temperature in Celsius scale and the corresponding fan speed in percentage are concurrently displayed on an LCD panel, ensuring user-friendly interaction with the system.

Materials

  • Arduino UNO Board
  • LM35 Temperature Sensor
  • 12V DC Fan
  • 16x2 LCD Display
  • Resistor 1k
  • LED 5mm
  • 12V Power Supply/Adapter
  • Connecting Wires
  • Breadboard

Code

#include
LiquidCrystal lcd(2,3,4,5,6,7);
int tempPin = A1; // the output pin of LM35
int ledfan = 8; // led and fan pin
int temp;
int tempMin = 30; // the temperature to start the fan 50% whih is 35C
int tempMax = 40; // the maximum temperature is at 100% which is 40C
int fanSpeed;
int fanLCD;
void setup() {
pinMode(ledfan, OUTPUT);
pinMode(tempPin, INPUT);
lcd.begin(16,2);
Serial.begin(9600);
}
void loop()
{
temp = readTemp(); // get the temperature
Serial.println( temp );
if(temp < tempMin) // if temp is lower than minimum temp
{
fanSpeed = 0; // fan is not spinning
digitalWrite(ledfan, fanSpeed);
fanLCD=0;
digitalWrite(ledfan, LOW);
}
if((temp >= tempMin) && (temp <= tempMax)) // if temperature is higher than minimum temp
{
fanLCD = map(temp, tempMin, tempMax, 0, 100);
}
if(temp > tempMax - 6) // if temp is at %50 start fan
{
digitalWrite(ledfan, HIGH); // turn on led and fan
}
else // else turn of led
{
digitalWrite(ledfan, LOW);
}
lcd.print("TEMP: ");
lcd.print(temp); // display the temperature
lcd.print("C ");
lcd.setCursor(0,1); // move cursor to next line
lcd.print("PEAK TEMP: ");
lcd.print(fanLCD); // display the fan speed
lcd.print("%");
delay(1000);
lcd.clear();
}
int readTemp() { // get the temperature and convert it to celsius
temp = analogRead(tempPin);
return temp * 0.48828125; }
